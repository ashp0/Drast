package Drast
version 0.2.0
default drast

target drast
	kind binary
	entry src/main.drast
	output build/bin/drast
	generated build/generated/drast
	include .
	cxx c++17

target test
	kind command
	depends drast
	command tests/run_tests.sh {output:drast} {root}

target install
	kind command
	depends drast
	command cp {output:drast} /usr/local/bin/drast
